 41class D4RLDataset(RLDataset):
 42    def __init__(self, d4rl_dset: dict[str, np.ndarray]):
 43        super().__init__()
 44        self.d4rl_dset = d4rl_dset
 45        self.episode_ends = np.where(d4rl_dset["terminals"] | d4rl_dset["timeouts"])[0]
 46        self.ep_lens = self.episode_ends[1:] - self.episode_ends[:-1]
 47        self.max_ep_len = self.ep_lens.max()
 48
 49    def get_description(self) -> str:
 50        return "D4RL"
 51
 52    @property
 53    def save_new_trajs_to(self):
 54        # disables saving new amago trajectories to disk
 55        return None
 56
 57    def sample_random_trajectory(self):
 58        episode_idx = random.randrange(0, len(self.episode_ends) - 1)
 59        return self._sample_trajectory(episode_idx)
 60
 61    def _sample_trajectory(self, episode_idx: int):
 62        # pick a random episode
 63        s = self.episode_ends[episode_idx] + 1
 64        e = self.episode_ends[episode_idx + 1] + 1
 65        traj_len = e - s
 66        obs_np = self.d4rl_dset["observations"][s : e + 1]
 67        actions_np = self.d4rl_dset["actions"][s:e]
 68        rewards_np = self.d4rl_dset["rewards"][s:e]
 69        terminals_np = self.d4rl_dset["terminals"][s:e]
 70        timeouts_np = self.d4rl_dset["timeouts"][s:e]
 71
 72        # convert to torch, adding time_idxs
 73        obs = {"observation": torch.from_numpy(obs_np)}
 74        actions = torch.from_numpy(actions_np).float()
 75        rewards = torch.from_numpy(rewards_np).float().unsqueeze(-1)
 76        time_idxs = torch.arange(traj_len).unsqueeze(-1).long()
 77        dones = torch.from_numpy(terminals_np).bool().unsqueeze(-1)
 78
 79        return RLData(
 80            obs=obs,
 81            actions=actions,
 82            rews=rewards,
 83            dones=dones,
 84            time_idxs=time_idxs,
 85        )

 92class D4RLGymEnv(gym.Env):
 93    """
 94    Light wrapper that logs the D4RL normalized return and handles
 95    the gym/gymnasium conversion while we're at it.
 96    """
 97
 98    def __init__(self, env_name: str):
 99        # hack fix seeding for parallel envs
100        np.random.seed(random.randrange(1e6))
101        self.env_name = env_name
102        self.env = og_gym.make(env_name)
103        self.action_space = space_convert(self.env.action_space)
104        self.observation_space = space_convert(self.env.observation_space)
105        if isinstance(self.env, og_gym.wrappers.TimeLimit):
106            # this time limit is apparently not consistent with the datasets
107            self.time_limit = self.env._max_episode_steps
108        else:
109            self.time_limit = None
110        self.max_return = d4rl.infos.REF_MAX_SCORE[self.env_name]
111        self.min_return = d4rl.infos.REF_MIN_SCORE[self.env_name]
112
113    @property
114    def dset(self):
115        return self.env.get_dataset()
116
117    def reset(self, *args, **kwargs):
118        self.episode_return = 0
119        return self.env.reset(), {}
120
121    def step(self, action):
122        s, r, d, i = self.env.step(action)
123        self.episode_return += r
124        if d:
125            i[f"{AMAGO_ENV_LOG_PREFIX} D4RL Normalized Return"] = (
126                d4rl.get_normalized_score(self.env_name, self.episode_return)
127            )
128        return s, r, d, d, i
